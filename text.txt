import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        String[] parts = input.nextLine().trim().split("\\s+");

        StringBuilder pin = new StringBuilder();

        for (String num : parts) {
            int reducedDigit = reduceToSingleDigit(num);

            if (reducedDigit % 2 == 1) {
                pin.append((char) ('a' + reducedDigit - 1));
            } else {
                pin.append(reducedDigit);
            }
        }

        System.out.println(pin.toString());
    }

    static int reduceToSingleDigit(String num) {
        int sum = 0;

        for (char c : num.toCharArray()) {
            if (Character.isDigit(c)) {
                sum += c - '0';
            }
        }

        while (sum > 9) {
            int temp = 0;
            while (sum > 0) {
                temp += sum % 10;
                sum /= 10;
            }
            sum = temp;
        }

        return sum;
    }
}
